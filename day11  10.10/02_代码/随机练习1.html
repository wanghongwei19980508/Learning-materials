var oDiv = document.querySelector('div'); oDiv.innerHTML = getNowTimeStr(); oDiv.style.color = setColor(); setInterval( function(){ oDiv.innerHTML = getNowTimeStr(); oDiv.style.color = setColor(); } , 1000 ) function getNowTimeStr(){ var time = new Date(); var year = time.getFullYear(); var month = time.getMonth() + 1; var date = time.getDate(); var weekArr = ['日','一','二','三','四','五','六']; var week = `星期${weekArr[time.getDay()]}`; var hours = time.getHours() < 10 ? '0'+time.getHours() : time.getHours(); var minutes = time.getMinutes() < 10 ? '0'+time.getMinutes() : time.getMinutes(); var seconds = time.getSeconds() < 10 ? '0'+time.getSeconds() : time.getSeconds() ; return `当前时间是${year}年,${month}月,${date}日,${week},${hours}时,${minutes}分,${seconds}秒`; }
